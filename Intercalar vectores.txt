void cargar_A(int A[], int t1)
{
	int aux,j;
	for (int i = 0; i <t1; i++)
	{
		cout << "v[" << i + 1 << "]=";
		cin >> A[i];
	}
	for (int i = 0; i<t1; i++) {
		for (j = 0; j<t1 - 1; j++) {
			if (A[j]>A[j + 1]) {
				aux = A[j];
				A[j] = A[j + 1];
				A[j + 1] = aux;
			}
		}
	}
}
